# STEP 1: Backup .env files
mkdir -p ~/backups/env_$(date +%Y%m%d_%H%M%S)
BACKUP_DIR=~/backups/env_$(date +%Y%m%d_%H%M%S)
cp /home/brian/whitelight/whitelight-backend/.env $BACKUP_DIR/backend.env
cp /home/brian/whitelight/whitelight/.env $BACKUP_DIR/frontend.env
echo "Backup saved to: $BACKUP_DIR"

# STEP 2: Pull latest code
cd /home/brian/whitelight

# STEP 3: Update backend dependencies
cd /home/brian/whitelight/whitelight-backend

# STEP 4: Restart backend
pm2 restart all

# STEP 5: Verify
pm2 list
